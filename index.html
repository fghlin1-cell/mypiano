// 1. 建立一個全域變數來儲存 AudioContext let audioCtx = null; // 2. 修復：針對 iOS 的「靜音解鎖」小技巧 function unlockAudioContext(context) { if (context.state !== 'suspended') return; const b = document.body; const events = ['touchstart', 'touchend', 'mousedown', 'keydown']; events.forEach(e => b.addEventListener(e, unlock, { once: true })); function unlock() { context.resume().then(clean); } function clean() { events.forEach(e => b.removeEventListener(e, unlock)); } } // 3. 啟動 App 的核心邏輯 async function startApp() { // 建立 AudioContext (相容性寫法) const AudioContextClass = window.AudioContext || window.webkitAudioContext; if (!audioCtx) { audioCtx = new AudioContextClass(); } // 強制啟動 if (audioCtx.state === 'suspended') { await audioCtx.resume(); } // 額外執行一次空的音效來「熱身」硬體 const oscillator = audioCtx.createOscillator(); const gain = audioCtx.createGain(); gain.gain.value = 0; // 無聲 oscillator.connect(gain); gain.connect(audioCtx.destination); oscillator.start(0); oscillator.stop(0.1); document.getElementById('overlay').style.display = 'none'; }
